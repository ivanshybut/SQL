# SQL

```sql
 SELECT - Извлекает данные из базы данных / extracts data from a database 
 UPDATE - Обнавляет данные в базе / updates data in a database
 DELETE - Удаляет данные из базы данных / deletes data from a database
 INSERT INTO - Вставляет новые данные в бау данных / inserts new data a database
 CREATE DATABASE - Создает новую базу данных / creates a new database
 ALTER DATABASE - Изменяет базу данных / modifles a database
 CREATE TABLE - Создает новую таблицу данных / creates a new table
 ALTER TABLE - Изменяет таблицу данных / modifles a table
 DROP TABLE - Удаляет таблицу / deletes a table
 CREATE INDEX - Создает индекс (ключ поиска) / creates an index (search key)
 DROP INDEX - Удаляет индекс / deletes an index
```
